```
//  功能:测试算术运算及表达式

#import <Foundation/Foundation.h>
#import "ArithExpress.h"
#import "Calculator.h"
int main(int argc, const char * argv[])
{

    @autoreleasepool {
        
        
        NSLog(@"====例1:测试算术运算=====");
        ArithExpress *ari = [[ArithExpress alloc]init];
        [ari testDataTypeConstans];
        [ari testPriority];
        [ari testIntegerArith];
        [ari testModulo];
        [ari testInteger2Float];
        [ari testEq];
        
        NSLog(@"====例2:计算器=====");
        Calculator *deskCalc = [[Calculator alloc] init];
        [deskCalc clear];
        [deskCalc setAccumulator:100.0];
        [deskCalc add:200.0];
        [deskCalc divide:15.0];
        [deskCalc subtract:10.0];
        [deskCalc multiply:5];
        NSLog(@"结果: %g",[deskCalc accumulator]);
        
    }
    return 0;
}
```

## Calculator.m

```
//
//  Calculator.m
//  BasicTest
//  功能：进行加减乘除运算
//  Created by bixiaopeng on 14-2-2.
//  Copyright (c) 2014年 bixiaopeng. All rights reserved.
//

#import "Calculator.h"

@implementation Calculator

//赋值
-(void) setAccumulator:(double)value
{
    accumulator = value;
}

//清空
-(void) clear
{
    accumulator = 0;
}

//返回值
-(double) accumulator
{
    return accumulator;
}

//加
-(void) add: (double) value
{
    accumulator += value;
}

//减
-(void) subtract:(double)value
{
    accumulator -= value;
}

//乘
-(void) multiply: (double) value
{
    accumulator *= value;
}

//除
-(void) divide:(double)value
{
    accumulator /=value;
}

@end
```
